001package io.prometheus.client.exporter;
002
003import java.io.IOException;
004import java.io.Writer;
005
006import javax.servlet.ServletException;
007import javax.servlet.http.HttpServlet;
008import javax.servlet.http.HttpServletRequest;
009import javax.servlet.http.HttpServletResponse;
010
011import io.prometheus.client.CollectorRegistry;
012import io.prometheus.client.exporter.common.TextFormat;
013
014public class MetricsServlet extends HttpServlet {
015
016  private CollectorRegistry registry;
017
018  /**
019   * Construct a MetricsServlet for the default registry.
020   */
021  public MetricsServlet() {
022    this(CollectorRegistry.defaultRegistry);
023  }
024
025  /**
026   * Construct a MetricsServlet for the given registry.
027   */
028  public MetricsServlet(CollectorRegistry registry) {
029    this.registry = registry;
030  }
031
032  @Override
033  protected void doGet(final HttpServletRequest req, final HttpServletResponse resp)
034      throws ServletException, IOException {
035    resp.setStatus(HttpServletResponse.SC_OK);
036    resp.setContentType(TextFormat.CONTENT_TYPE_004);
037
038    Writer writer = resp.getWriter();
039    try {
040      TextFormat.write004(writer, registry.metricFamilySamples());
041      writer.flush();
042    } finally {
043      writer.close();
044    }
045  }
046
047  @Override
048  protected void doPost(final HttpServletRequest req, final HttpServletResponse resp)
049      throws ServletException, IOException {
050    doGet(req, resp);
051  }
052
053}
