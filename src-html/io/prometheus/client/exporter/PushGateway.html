020/**
021 * Export metrics via the Prometheus Pushgateway.
022 * <p>
023 * The Prometheus Pushgateway exists to allow ephemeral and batch jobs to expose their metrics to Prometheus.
024 * Since these kinds of jobs may not exist long enough to be scraped, they can instead push their metrics
025 * to a Pushgateway. This class allows pushing the contents of a {@link CollectorRegistry} to
026 * a Pushgateway.
027 * <p>
028 * Example usage:
029 * <pre>
030 * {@code
031 *   void executeBatchJob() throws Exception {
032 *     CollectorRegistry registry = new CollectorRegistry();
033 *     Gauge duration = Gauge.build()
034 *         .name("my_batch_job_duration_seconds").help("Duration of my batch job in seconds.").register(registry);
035 *     Gauge.Timer durationTimer = duration.startTimer();
036 *     try {
037 *       // Your code here.
038 *
039 *       // This is only added to the registry after success,
040 *       // so that a previous success in the Pushgateway isn't overwritten on failure.
041 *       Gauge lastSuccess = Gauge.build()
042 *           .name("my_batch_job_last_success").help("Last time my batch job succeeded, in unixtime.").register(registry);
043 *       lastSuccess.setToCurrentTime();
044 *     } finally {
045 *       durationTimer.setDuration();
046 *       PushGateway pg = new PushGateway("127.0.0.1:9091");
047 *       pg.pushAdd(registry, "my_batch_job");
048 *     }
049 *   }
050 * }
051 * </pre>
052 * <p>
053 * See <a href="https://github.com/prometheus/pushgateway">https://github.com/prometheus/pushgateway</a>
054 */
055public class PushGateway {
056
057  private final String address;

059  /**

237  void doRequest(CollectorRegistry registry, String job, Map<String, String> groupingKey, String method) throws IOException {
238    String url = "http://" + address + "/metrics/job/" + URLEncoder.encode(job, "UTF-8");
239    if (groupingKey != null) {
240      for (Map.Entry<String, String> entry: groupingKey.entrySet()) {
241        url += "/" + entry.getKey() + "/" + URLEncoder.encode(entry.getValue(), "UTF-8");
242      }
243    }
244    HttpURLConnection connection = (HttpURLConnection) new URL(url).openConnection();
245    connection.setRequestProperty("Content-Type", TextFormat.CONTENT_TYPE_004);
246    if (!method.equals("DELETE")) {
247      connection.setDoOutput(true);
248    }
249    connection.setRequestMethod(method);
250
251    connection.setConnectTimeout(10 * SECONDS_PER_MILLISECOND);
252    connection.setReadTimeout(10 * SECONDS_PER_MILLISECOND);
253    connection.connect();
254
255    try {
256      if (!method.equals("DELETE")) {
257        BufferedWriter writer = new BufferedWriter(new OutputStreamWriter(connection.getOutputStream(), "UTF-8"));
258        TextFormat.write004(writer, registry.metricFamilySamples());
259        writer.flush();
260        writer.close();
261      }
262
263      int response = connection.getResponseCode();
264      if (response != HttpURLConnection.HTTP_ACCEPTED) {
265        throw new IOException("Response code from " + url + " was " + response);
266      }
267    } finally {
268      connection.disconnect();
269    }
270  }

272  /**
273   * Returns a grouping key with the instance label set to the machine's IP address.
274   * <p>
275   * This is a convenience function, and should only be used where you want to
276   * push per-instance metrics rather than cluster/job level metrics.
277  */
278  public static Map<String, String> instanceIPGroupingKey() throws UnknownHostException {
279    Map<String, String> groupingKey = new HashMap<String, String>();
280    groupingKey.put("instance", InetAddress.getLocalHost().getHostAddress());
281    return groupingKey;
282  }
