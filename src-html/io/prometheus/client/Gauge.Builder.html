009/**
010 * Gauge metric, to report instantaneous values.
011 * <p>
012 * Examples of Gauges include:
013 * <ul>
014 *  <li>Inprogress requests</li>
015 *  <li>Number of items in a queue</li>
016 *  <li>Free memory</li>
017 *  <li>Total memory</li>
018 *  <li>Temperature</li>
019 * </ul>
020 *
021 * Gauges can go both up and down.
022 * <p>
023 * An example Gauge:
024 * <pre>
025 * {@code
026 *   class YourClass {
027 *     static final Gauge inprogressRequests = Gauge.build()
028 *         .name("inprogress_requests").help("Inprogress requests.").register();
029 *
030 *     void processRequest() {
031 *        inprogressRequest.inc();
032 *        // Your code here.
033 *        inprogressRequest.dec();
034 *     }
035 *   }
036 * }
037 * </pre>
038 *
039 * <p>
040 * You can also use labels to track different types of metric:
041 * <pre>
042 * {@code
043 *   class YourClass {
044 *     static final Gauge inprogressRequests = Gauge.build()
045 *         .name("inprogress_requests").help("Inprogress requests.")
046 *         .labelNames("method").register();
047 *
048 *     void processGetRequest() {
049 *        inprogressRequests.labels("get").inc();
050 *        // Your code here.
051 *        inprogressRequests.labels("get").dec();
052 *     }
053 *     void processPostRequest() {
054 *        inprogressRequests.labels("post").inc();
055 *        // Your code here.
056 *        inprogressRequests.labels("post").dec();
057 *     }
058 *   }
059 * }
060 * </pre>
061 * <p>
062 * These can be aggregated and processed together much more easily in the Prometheus
063 * server than individual metrics for each labelset.
064 */
065public class Gauge extends SimpleCollector<Gauge.Child> implements Collector.Describable {
066

090   /**
091    * Represents an event being timed.
092    */
093   public static class Timer implements Closeable {
094     private final Child child;
095     private final long start;
096     private Timer(Child child) {
097       this.child = child;
098       start = Child.timeProvider.nanoTime();
099     }
100     /**
101      * Set the amount of time in seconds since {@link Child#startTimer} was called.
102      * @return Measured duration in seconds since {@link Child#startTimer} was called.
103      */
104     public double setDuration() {
105       double elapsed = (Child.timeProvider.nanoTime() - start) / NANOSECONDS_PER_SECOND;
106       child.set(elapsed);
107       return elapsed;
108     }
109
110     /**
111      * Equivalent to calling {@link #setDuration()}.
112      * @throws IOException
113      */
114     @Override
115     public void close() throws IOException {
116       setDuration();
117     }
118   }
119
120  /**
121   * The value of a single Gauge.
122   * <p>
123   * <em>Warning:</em> References to a Child become invalid after using
124   * {@link SimpleCollector#remove} or {@link SimpleCollector#clear},
125   */
126  public static class Child {
127    private final DoubleAdder value = new DoubleAdder();
128
129    static TimeProvider timeProvider = new TimeProvider();
130    /**
131     * Increment the gauge by 1.
132     */
133    public void inc() {
134      inc(1);
135    }
136    /**
137     * Increment the gauge by the given amount.
138     */
139    public void inc(double amt) {
140      value.add(amt);
141    }
142    /**
143     * Decrement the gauge by 1.
144     */
145    public void dec() {
146      dec(1);
147    }
148    /**
149     * Decrement the gauge by the given amount.
150     */
151    public void dec(double amt) {
152      value.add(-amt);
153    }
154    /**
155     * Set the gauge to the given value.
156     */
157    public void set(double val) {
158      synchronized(this) {
159        value.reset();
160        // If get() were called here it'd see an invalid value, so use a lock.
161        // inc()/dec() don't need locks, as all the possible outcomes
162        // are still possible if set() were atomic so no new races are introduced.
163        value.add(val);
164      }
165    }
166    /**
167     * Set the gauge to the current unixtime.
168     */
169    public void setToCurrentTime() {
170      set(timeProvider.currentTimeMillis() / MILLISECONDS_PER_SECOND);
171    }
172    /**
173     * Start a timer to track a duration.
174     * <p>
175     * Call {@link Timer#setDuration} at the end of what you want to measure the duration of.
176     * <p>
177     * This is primarily useful for tracking the durations of major steps of batch jobs,
178     * which are then pushed to a PushGateway.
179     * For tracking other durations/latencies you should usually use a {@link Summary}.
180     */
181    public Timer startTimer() {
182      return new Timer(this);
183    }
184
185    /**
186     * Executes runnable code (i.e. a Java 8 Lambda) and observes a duration of how long it took to run.
187     *
188     * @param timeable Code that is being timed
189     * @return Measured duration in seconds for timeable to complete.
190     */
191    public double setToTime(Runnable timeable){
192      Timer timer = startTimer();
193
194      double elapsed;
195      try {
196        timeable.run();
197      } finally {
198        elapsed = timer.setDuration();
199      }
200
201      return elapsed;
202    }
203
204    /**
205     * Get the value of the gauge.
206     */
207    public double get() {
208      synchronized(this) {
209        return value.sum();
210      }
211    }
212  }

274  @Override
275  public List<MetricFamilySamples> collect() {
276    List<MetricFamilySamples.Sample> samples = new ArrayList<MetricFamilySamples.Sample>();
277    for(Map.Entry<List<String>, Child> c: children.entrySet()) {
278      samples.add(new MetricFamilySamples.Sample(fullname, labelNames, c.getKey(), c.getValue().get()));
279    }
280    MetricFamilySamples mfs = new MetricFamilySamples(fullname, Type.GAUGE, help, samples);
281
282    List<MetricFamilySamples> mfsList = new ArrayList<MetricFamilySamples>();
283    mfsList.add(mfs);
284    return mfsList;
285  }
286
287  @Override
288  public List<MetricFamilySamples> describe() {
289    List<MetricFamilySamples> mfsList = new ArrayList<MetricFamilySamples>();
290    mfsList.add(new GaugeMetricFamily(fullname, help, labelNames));
291    return mfsList;
292  }

294  static class TimeProvider {
295    long currentTimeMillis() {
296      return System.currentTimeMillis();
297    }
298    long nanoTime() {
299      return System.nanoTime();
300    }
301  }
