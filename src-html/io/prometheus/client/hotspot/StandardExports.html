001package io.prometheus.client.hotspot;
002
003import com.sun.management.OperatingSystemMXBean;
004import com.sun.management.UnixOperatingSystemMXBean;
005import java.io.BufferedReader;
006import java.io.FileReader;
007import java.io.IOException;
008import java.io.FileNotFoundException;
009import java.lang.management.RuntimeMXBean;
010import java.lang.management.ManagementFactory;
011import java.util.ArrayList;
012import java.util.Collections;
013import java.util.List;
014import java.util.logging.Logger;
015
016import io.prometheus.client.Collector;
017import io.prometheus.client.CounterMetricFamily;
018import io.prometheus.client.GaugeMetricFamily;
019
020/**
021 * Exports the standard exports common across all prometheus clients.
022 * <p>
023 * This includes stats like CPU time spent and memory usage.
024 * <p>
025 * Example usage:
026 * <pre>
027 * {@code
028 *   new StandardExports().register();
029 * }
030 * </pre>
031 */
032public class StandardExports extends Collector {
033  private static final Logger LOGGER = Logger.getLogger(StandardExports.class.getName());
034
035  private final StatusReader statusReader;
036  private final OperatingSystemMXBean osBean;
037  private final RuntimeMXBean runtimeBean;
038  private final boolean unix;
039  private final boolean linux;
040
041  public StandardExports() {
042    this(new StatusReader(),
043         (OperatingSystemMXBean)ManagementFactory.getOperatingSystemMXBean(),
044         ManagementFactory.getRuntimeMXBean());
045  }
046
047  StandardExports(StatusReader statusReader, OperatingSystemMXBean osBean, RuntimeMXBean runtimeBean) {
048      this.statusReader = statusReader;
049      this.osBean = osBean;
050      this.runtimeBean = runtimeBean;
051      this.unix = (osBean instanceof UnixOperatingSystemMXBean);
052      this.linux = (osBean.getName().indexOf("Linux") == 0);
053  }
054
055  private final static double KB = 1024;
056
057  public List<MetricFamilySamples> collect() {
058    List<MetricFamilySamples> mfs = new ArrayList<MetricFamilySamples>();
059
060    mfs.add(new CounterMetricFamily("process_cpu_seconds_total", "Total user and system CPU time spent in seconds.",
061        osBean.getProcessCpuTime() / NANOSECONDS_PER_SECOND));
062
063    mfs.add(new GaugeMetricFamily("process_start_time_seconds","Start time of the process since unix epoch in seconds.",
064        runtimeBean.getStartTime() / MILLISECONDS_PER_SECOND));
065
066    if (unix) {
067      UnixOperatingSystemMXBean unixBean = (UnixOperatingSystemMXBean) osBean;
068      mfs.add(new GaugeMetricFamily("process_open_fds", "Number of open file descriptors.",
069          unixBean.getOpenFileDescriptorCount()));
070      mfs.add(new GaugeMetricFamily("process_max_fds", "Maximum number of open file descriptors.",
071          unixBean.getMaxFileDescriptorCount()));
072    }
073
074    // There's no standard Java or POSIX way to get memory stats,
075    // so add support for just Linux for now.
076    if (linux) {
077      try {
078      collectMemoryMetricsLinux(mfs);
079      } catch (Exception e) {
080        // If the format changes, log a warning and return what we can.
081        LOGGER.warning(e.toString());
082      }
083    }
084    return mfs;
085  }
086
087  void collectMemoryMetricsLinux(List<MetricFamilySamples> mfs) {
088    // statm/stat report in pages, and it's non-trivial to get pagesize from Java
089    // so we parse status instead.
090    BufferedReader br = null;
091    try {
092      br = statusReader.procSelfStatusReader();
093      String line;
094      while ((line = br.readLine()) != null) {
095        if (line.startsWith("VmSize:")) {
096          mfs.add(new GaugeMetricFamily("process_virtual_memory_bytes",
097              "Virtual memory size in bytes.",
098              Float.parseFloat(line.split("\\s+")[1]) * KB));
099        } else if (line.startsWith("VmRSS:")) {
100          mfs.add(new GaugeMetricFamily("process_resident_memory_bytes",
101              "Resident memory size in bytes.",
102              Float.parseFloat(line.split("\\s+")[1]) * KB));
103        }
104      }
105    } catch (IOException e) {
106      LOGGER.fine(e.toString());
107    } finally {
108      if (br != null) {
109        try {
110          br.close();
111        } catch (IOException e) {
112          LOGGER.fine(e.toString());
113        }
114      }
115    }
116  }
117
118  static class StatusReader {
119    BufferedReader procSelfStatusReader() throws FileNotFoundException {
120      return new BufferedReader(new FileReader("/proc/self/status"));
121    }
122  }
123}
