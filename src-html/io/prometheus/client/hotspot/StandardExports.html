001package io.prometheus.client.hotspot;
002
003import com.sun.management.OperatingSystemMXBean;
004import com.sun.management.UnixOperatingSystemMXBean;
005import java.io.BufferedReader;
006import java.io.FileReader;
007import java.io.IOException;
008import java.io.FileNotFoundException;
009import java.lang.management.RuntimeMXBean;
010import java.lang.management.ManagementFactory;
011import java.util.ArrayList;
012import java.util.Collections;
013import java.util.List;
014import java.util.logging.Logger;
015
016import io.prometheus.client.Collector;
017
018/**
019 * Exports the standard exports common across all prometheus clients.
020 * <p>
021 * This includes stats like CPU time spent and memory usage.
022 * <p>
023 * Example usage:
024 * <pre>
025 * {@code
026 *   new StandardExports().register();
027 * }
028 * </pre>
029 */
030public class StandardExports extends Collector {
031  private static final Logger LOGGER = Logger.getLogger(StandardExports.class.getName());
032
033  private final StatusReader statusReader;
034  private final OperatingSystemMXBean osBean;
035  private final RuntimeMXBean runtimeBean;
036  private final boolean unix;
037  private final boolean linux;
038
039  public StandardExports() {
040    this(new StatusReader(),
041         (OperatingSystemMXBean)ManagementFactory.getOperatingSystemMXBean(),
042         ManagementFactory.getRuntimeMXBean());
043  }
044
045  StandardExports(StatusReader statusReader, OperatingSystemMXBean osBean, RuntimeMXBean runtimeBean) {
046      this.statusReader = statusReader;
047      this.osBean = osBean;
048      this.runtimeBean = runtimeBean;
049      this.unix = (osBean instanceof UnixOperatingSystemMXBean);
050      this.linux = (osBean.getName().indexOf("Linux") == 0);
051  }
052
053  private static MetricFamilySamples singleMetric(String name, Type type, String help, double value) {
054    List<MetricFamilySamples.Sample> samples = Collections.singletonList(
055        new MetricFamilySamples.Sample(name, Collections.<String>emptyList(), Collections.<String>emptyList(), value));
056    return new MetricFamilySamples(name, type, help, samples);
057  }
058
059  private final static double KB = 1024;
060
061  public List<MetricFamilySamples> collect() {
062    List<MetricFamilySamples> mfs = new ArrayList<MetricFamilySamples>();
063
064    mfs.add(singleMetric("process_cpu_seconds_total", Type.COUNTER, "Total user and system CPU time spent in seconds.",
065        osBean.getProcessCpuTime() / NANOSECONDS_PER_SECOND));
066
067    mfs.add(singleMetric("process_start_time_seconds", Type.GAUGE, "Start time of the process since unix epoch in seconds.",
068        runtimeBean.getStartTime() / MILLISECONDS_PER_SECOND));
069
070    if (unix) {
071      UnixOperatingSystemMXBean unixBean = (UnixOperatingSystemMXBean) osBean;
072      mfs.add(singleMetric("process_open_fds", Type.GAUGE, "Number of open file descriptors.",
073          unixBean.getOpenFileDescriptorCount()));
074      mfs.add(singleMetric("process_max_fds", Type.GAUGE, "Maximum number of open file descriptors.",
075          unixBean.getMaxFileDescriptorCount()));
076    }
077
078    // There's no standard Java or POSIX way to get memory stats,
079    // so add support for just Linux for now.
080    if (linux) {
081      try {
082      collectMemoryMetricsLinux(mfs);
083      } catch (Exception e) {
084        // If the format changes, log a warning and return what we can.
085        LOGGER.warning(e.toString());
086      }
087    }
088    return mfs;
089  }
090
091  void collectMemoryMetricsLinux(List<MetricFamilySamples> mfs) {
092    // statm/stat report in pages, and it's non-trivial to get pagesize from Java
093    // so we parse status instead.
094    BufferedReader br = null;
095    try {
096      br = statusReader.procSelfStatusReader();
097      String line;
098      while ((line = br.readLine()) != null) {
099        if (line.startsWith("VmSize:")) {
100          mfs.add(singleMetric("process_virtual_memory_bytes", Type.GAUGE,
101              "Virtual memory size in bytes.",
102              Float.parseFloat(line.split("\\s+")[1]) * KB));
103        } else if (line.startsWith("VmRSS:")) {
104          mfs.add(singleMetric("process_resident_memory_bytes", Type.GAUGE,
105              "Resident memory size in bytes.",
106              Float.parseFloat(line.split("\\s+")[1]) * KB));
107        }
108      }
109    } catch (IOException e) {
110      LOGGER.fine(e.toString());
111    } finally {
112      if (br != null) {
113        try {
114          br.close();
115        } catch (IOException e) {
116          LOGGER.fine(e.toString());
117        }
118      }
119    }
120  }
121
122  static class StatusReader {
123    BufferedReader procSelfStatusReader() throws FileNotFoundException {
124      return new BufferedReader(new FileReader("/proc/self/status"));
125    }
126  }
127}
